### B. [CSP-J 2025] 座位 / seat（民间数据）

#### 做法 1

```cpp
#include <bits/stdc++.h>
using namespace std;
// n 行 m 列
int n, m;
int a[105];
// 开个二维数组真的填进去
int b[15][15];
bool cmp(int x, int y)
{
    return x > y;
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> m;
    for (int i = 1; i <= n * m; i++)
        cin >> a[i];
    // 记住小 R 的得分，排序
    int R = a[1];
    sort(a + 1, a + n * m + 1, cmp);
    // 填数字
    // 第一列 -> 最后一列
    int now = 1;
    for (int j = 1; j <= m; j++)
    {
        // 奇数列：1 -> n 行，偶数列：n -> 1 行
        if (j % 2 == 1)
        {
            for (int i = 1; i <= n; i++)
            {
                b[i][j] = a[now];
                now++;
            }
        }
        else
        {
            for (int i = n; i >= 1; i--)
            {
                b[i][j] = a[now];
                now++;
            }
        }
    }
    // 找找 R 在哪儿
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            if (b[i][j] == R)
                cout << j << " " << i << "\n";
    return 0;
}
```

### C. [CSP-J 2025] 异或和 / xor（民间数据）

#### 经济实惠快速的 $60$ 分

```cpp
#include <bits/stdc++.h>
using namespace std;
const int MAXN = 500000;
int n, k;
int a[MAXN + 5];
int sum[MAXN + 5];
// 前 i 个数能凑出来的组数
int f[MAXN + 5];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> k;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    // 处理前缀异或和
    sum[0] = 0;
    for (int i = 1; i <= n; i++)
        sum[i] = sum[i - 1] ^ a[i];
    // dp
    f[0] = 0;
    for (int i = 1; i <= n; i++)
    {
        f[i] = f[i - 1];
        // 找到一个最近的 j 使得 a[j]~a[i] 异或和为 k
        for (int j = i; j >= 1; j--)
            if ((sum[i] ^ sum[j - 1]) == k)
                // a[1]~a[j-1] 的最多数量 + 一组（a[j]~a[i]）
                f[i] = max(f[i], f[j - 1] + 1);
    }
    cout << f[n];
    return 0;
}
```

### [CSP-J 2025] 多边形 / polygon（民间数据）

#### 经济实惠快速的 $64$ 分

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int MOD = 998244353;
int n, ans;
int a[5005];
// 现在考虑第 now 个数，已经选择了 cnt 个数
// 这些数之和是 sum，最大值是 maxA
void dfs(int now, int cnt, int sum, int maxA)
{
    if (now > n)
    {
        if (sum > 2 * maxA)
            ans++;
        return;
    }
    dfs(now + 1, cnt, sum, maxA);
    dfs(now + 1, cnt + 1, sum + a[now], max(maxA, a[now]));
}
signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    sort(a + 1, a + n + 1);
    if (n == 3)
    {
        if (a[1] + a[2] > a[3])
            cout << 1;
        else
            cout << 0;
    }
    else if (a[n] == 1)
    {
        // 2^n - C(n,0) - C(n,1) - C(n,2)
        int ans = 1;
        for (int i = 1; i <= n; i++)
            ans = ans * 2 % MOD;
        ans -= 1;
        ans -= n;
        ans -= n * (n - 1) / 2;
        ans %= MOD; // -MOD+1 ~ MOD-1
        ans = (ans + MOD) % MOD;
        cout << ans;
    }
    else
    {
        dfs(1, 0, 0, 0);
        cout << ans;
    }
    return 0;
}
```

#### 不错的正解

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int MOD = 998244353;
int n, ans;
int a[5005];
// f[i][j] 前 i 个数，和大于 j 的方案数
// 因为所有数都大于 0，所以 f[i][<0] 等价于 f[i][0]
int f[5005][5005];
signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    sort(a + 1, a + n + 1);
    // f[0][~] = 0
    f[0][0] = 1;
    for (int i = 1; i <= n; i++)
        for (int j = 0; j <= 5001; j++)
        {
            f[i][j] = f[i - 1][j] +
                      f[i - 1][max(0LL, j - a[i])];
            f[i][j] %= MOD;
        }
    int ans = 0;
    for (int i = 1; i <= n; i++)
    {
        // a[1]~a[i-1] > a[i]
        // a[1]~a[i-1] >= a[i]+1
        ans += f[i - 1][a[i] + 1];
        ans %= MOD;
    }
    cout << ans;
    return 0;
}
```

### A. P14361 [CSP-S 2025] 社团招新 / club（民间数据）

```cpp
#include <bits/stdc++.h>
using namespace std;
int n, ans;
struct Per
{
    int a, b, c;
};
Per a[100000 + 5];
int cal(Per x)
{
    if (x.a > x.b && x.a > x.c)
        return x.a - max(x.b, x.c);
    if (x.b > x.c)
        return x.b - max(x.a, x.c);
    return x.c - max(x.a, x.b);
}
bool cmp(Per x, Per y)
{
    return cal(x) < cal(y);
}
vector<Per> v[3 + 1];
void work()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i].a >> a[i].b >> a[i].c;
    ans = 0;
    for (int i = 1; i <= 3; i++)
        v[i].clear();
    for (int i = 1; i <= n; i++)
    {
        if (a[i].a > a[i].b && a[i].a > a[i].c)
            v[1].push_back(a[i]), ans += a[i].a;
        else if (a[i].b > a[i].c)
            v[2].push_back(a[i]), ans += a[i].b;
        else
            v[3].push_back(a[i]), ans += a[i].c;
    }
    int pos = 0;
    for (int i = 1; i <= 3; i++)
    {
        if (v[i].size() > n / 2)
            pos = i;
    }
    if (pos != 0)
    {
        sort(v[pos].begin(), v[pos].end(), cmp);
        int num = v[pos].size() - n / 2;
        for (int i = 0; i < num; i++)
            ans -= cal(v[pos][i]);
    }
    cout << ans << "\n";
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin >> t;
    while (t--)
        work();
    return 0;
}
```

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int MAXN = 10000;
const int MAXK = 10;
const int INF = 1'000'000'000LL * MAXN;
const int MAXM = 1000000;
int n, m, k;

// <权值,<起点,终点>>
// 初始边，nk 条边
vector<pair<int, pair<int, int>>> e, ee;

{
    if (x == fa[x])
        return x;

}
void work(int sta)
{
    int now = 0;
    for (int i = n + 1; i <= n + k; i++)
        if (sta & (1 << (i - n - 1)))
        {
            ok[i] = true;
            now += cost[i];

        else
            ok[i] = false;
    nowE = e;
    for (int i = 0; i < ee.size(); i++)
    {
        int w = ee[i].first;
        int u = ee[i].second.first;
        int v = ee[i].second.second;
        if (ok[u] && ok[v])
```

### C. [CSP-S 2025] 谐音替换 / replace（民间数据）

#### 50 分

用字符串哈希替代检查能拿到更多分，用 ACAM 就能拿到满分了。

```cpp
#include <bits/stdc++.h>
using namespace std;
int n, Q;
struct SS
{
    // ABC->ADC：A#BD#C
    string s;
    int aLen, bLen, cLen;
};
vector<SS> rl; // 所有变换规则
// 输入并添加一条变换规则
string s1, s2, s3;
SS getOne()
{
    cin >> s1 >> s2;
    int l, r;
    for (int i = 0; i < s1.size(); i++)
        if (s1[i] != s2[i])
        {
            l = i;
            break;
        }
    for (int i = s1.size() - 1; i >= 0; i--)
        if (s1[i] != s2[i])
        {
            r = i;
            break;
        }
    // 0~l-1 s1[l~r] s2[l~r] r+1~s.size()-1
    s3 = "";
    for (int i = 0; i <= l - 1; i++)
        s3 += s1[i];
    s3 += "#";
    for (int i = l; i <= r; i++)
        s3 += s1[i];
    for (int i = l; i <= r; i++)
        s3 += s2[i];
    s3 += "#";
    for (int i = r + 1; i <= s1.size() - 1; i++)
        s3 += s1[i];
    return {s3, l, r - l + 1, s1.size() - 1 - r};
}
// 检查 x 是否是 y 的子串
bool check(SS &x, SS &y)
{
    if (x.aLen > y.aLen ||
        x.bLen != y.bLen ||
        x.cLen > y.cLen)
        return false;
    for (int i = 0; i < x.s.size(); i++)
    {
        int j = i - x.aLen + y.aLen;
        if (x.s[i] != y.s[j])
            return false;
    }
    return true;
}
// 每一组询问
void work()
{
    SS now = getOne();
    int ans = 0;
    for (int i = 0; i < rl.size(); i++)
    {
        if (check(rl[i], now))
            ans++;
    }
    cout << ans << "\n";
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> Q;
    for (int i = 1; i <= n; i++)
        rl.push_back(getOne());
    while (Q--)
        work();
    return 0;
}
```
