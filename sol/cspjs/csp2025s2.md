## A. P14361 [CSP-S 2025] 社团招新 / club（民间数据）

```cpp
#include <bits/stdc++.h>
using namespace std;
int n, ans;
struct Per
{
    int a, b, c;
};
Per a[100000 + 5];
int cal(Per x)
{
    if (x.a > x.b && x.a > x.c)
        return x.a - max(x.b, x.c);
    if (x.b > x.c)
        return x.b - max(x.a, x.c);
    return x.c - max(x.a, x.b);
}
bool cmp(Per x, Per y)
{
    return cal(x) < cal(y);
}
vector<Per> v[3 + 1];
void work()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i].a >> a[i].b >> a[i].c;
    ans = 0;
    for (int i = 1; i <= 3; i++)
        v[i].clear();
    for (int i = 1; i <= n; i++)
    {
        if (a[i].a > a[i].b && a[i].a > a[i].c)
            v[1].push_back(a[i]), ans += a[i].a;
        else if (a[i].b > a[i].c)
            v[2].push_back(a[i]), ans += a[i].b;
        else
            v[3].push_back(a[i]), ans += a[i].c;
    }
    int pos = 0;
    for (int i = 1; i <= 3; i++)
    {
        if (v[i].size() > n / 2)
            pos = i;
    }
    if (pos != 0)
    {
        sort(v[pos].begin(), v[pos].end(), cmp);
        int num = v[pos].size() - n / 2;
        for (int i = 0; i < num; i++)
            ans -= cal(v[pos][i]);
    }
    cout << ans << "\n";
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin >> t;
    while (t--)
        work();
    return 0;
}
```

## B. [CSP-S 2025] 道路修复 / road（民间数据）

### 暴力每次重新 Kruskal 32 分

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int MAXN = 10000;
const int MAXK = 10;
const int INF = 1'000'000'000LL * MAXN;
const int MAXM = 1000000;
int n, m, k;
int cost[MAXN + MAXK + 5];
// <权值,<起点,终点>>
// 初始边，nk 条边
vector<pair<int, pair<int, int>>> e, ee;
int ans;
vector<pair<int, pair<int, int>>> nowE; // 当前情况下的有效边
bool ok[MAXN + MAXK + 5];               // 标记哪些城市有效
int fa[MAXN + MAXK + 5];
int findFa(int x)
{
    if (x == fa[x])
        return x;
    return fa[x] = findFa(fa[x]);
}
void work(int sta)
{
    int now = 0;
    for (int i = n + 1; i <= n + k; i++)
        if (sta & (1 << (i - n - 1)))
        {
            ok[i] = true;
            now += cost[i];
        }
        else
            ok[i] = false;
    nowE = e;
    for (int i = 0; i < ee.size(); i++)
    {
        int w = ee[i].first;
        int u = ee[i].second.first;
        int v = ee[i].second.second;
        if (ok[u] && ok[v])
            nowE.push_back({w, {u, v}});
    }
    sort(nowE.begin(), nowE.end());
    for (int i = 1; i <= n + k; i++)
        fa[i] = i;
    for (int i = 0; i < nowE.size(); i++)
    {
        int w = nowE[i].first;
        int u = nowE[i].second.first;
        int v = nowE[i].second.second;
        int faU = findFa(u);
        int faV = findFa(v);
        if (faU != faV)
        {
            now += w;
            fa[faU] = faV;
        }
    }
    ans = min(ans, now);
}
signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> m >> k;
    for (int i = 1; i <= n + k; i++)
        ok[i] = true;
    for (int i = 1; i <= m; i++)
    {
        int u, v, w;
        cin >> u >> v >> w;
        e.push_back({w, {u, v}});
    }
    for (int i = n + 1; i <= n + k; i++)
    {
        cin >> cost[i];
        for (int j = 1; j <= n; j++)
        {
            int w;
            cin >> w;
            ee.push_back({w, {i, j}});
        }
    }
    ans = INF;
    for (int sta = 0; sta <= (1 << k) - 1; sta++)
        work(sta);
    cout << ans;
    return 0;
}
```

## C. [CSP-S 2025] 谐音替换 / replace（民间数据）

### 50 分

用字符串哈希替代检查能拿到更多分，用 ACAM 就能拿到满分了。

```cpp
#include <bits/stdc++.h>
using namespace std;
int n, Q;
struct SS
{
    // ABC->ADC：A#BD#C
    string s;
    int aLen, bLen, cLen;
};
vector<SS> rl; // 所有变换规则
// 输入并添加一条变换规则
string s1, s2, s3;
SS getOne()
{
    cin >> s1 >> s2;
    int l, r;
    for (int i = 0; i < s1.size(); i++)
        if (s1[i] != s2[i])
        {
            l = i;
            break;
        }
    for (int i = s1.size() - 1; i >= 0; i--)
        if (s1[i] != s2[i])
        {
            r = i;
            break;
        }
    // 0~l-1 s1[l~r] s2[l~r] r+1~s.size()-1
    s3 = "";
    for (int i = 0; i <= l - 1; i++)
        s3 += s1[i];
    s3 += "#";
    for (int i = l; i <= r; i++)
        s3 += s1[i];
    for (int i = l; i <= r; i++)
        s3 += s2[i];
    s3 += "#";
    for (int i = r + 1; i <= s1.size() - 1; i++)
        s3 += s1[i];
    return {s3, l, r - l + 1, s1.size() - 1 - r};
}
// 检查 x 是否是 y 的子串
bool check(SS &x, SS &y)
{
    if (x.aLen > y.aLen ||
        x.bLen != y.bLen ||
        x.cLen > y.cLen)
        return false;
    for (int i = 0; i < x.s.size(); i++)
    {
        int j = i - x.aLen + y.aLen;
        if (x.s[i] != y.s[j])
            return false;
    }
    return true;
}
// 每一组询问
void work()
{
    SS now = getOne();
    int ans = 0;
    for (int i = 0; i < rl.size(); i++)
    {
        if (check(rl[i], now))
            ans++;
    }
    cout << ans << "\n";
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> Q;
    for (int i = 1; i <= n; i++)
        rl.push_back(getOne());
    while (Q--)
        work();
    return 0;
}
```
