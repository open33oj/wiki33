## A. [CSP-J 2025] 拼数 / number（民间数据）

### 做法 1

```cpp
#include <bits/stdc++.h>
using namespace std;
string s;
string t;
bool cmp(char x, char y)
{
    return x > y;
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> s;
    t = "";
    for (int i = 0; i < s.size(); i++)
        if ('0' <= s[i] && s[i] <= '9')
            t += s[i]; // 注意如果 t = t + s[i]; 会超时
    sort(t.begin(), t.end(), cmp);
    cout << t;
    return 0;
}
```

## B. [CSP-J 2025] 座位 / seat（民间数据）

### 做法 1

```cpp
#include <bits/stdc++.h>
using namespace std;
// n 行 m 列
int n, m;
int a[105];
// 开个二维数组真的填进去
int b[15][15];
bool cmp(int x, int y)
{
    return x > y;
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> m;
    for (int i = 1; i <= n * m; i++)
        cin >> a[i];
    // 记住小 R 的得分，排序
    int R = a[1];
    sort(a + 1, a + n * m + 1, cmp);
    // 填数字
    // 第一列 -> 最后一列
    int now = 1;
    for (int j = 1; j <= m; j++)
    {
        // 奇数列：1 -> n 行，偶数列：n -> 1 行
        if (j % 2 == 1)
        {
            for (int i = 1; i <= n; i++)
            {
                b[i][j] = a[now];
                now++;
            }
        }
        else
        {
            for (int i = n; i >= 1; i--)
            {
                b[i][j] = a[now];
                now++;
            }
        }
    }
    // 找找 R 在哪儿
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            if (b[i][j] == R)
                cout << j << " " << i << "\n";
    return 0;
}
```

```cpp
    {
        if (a[i] == R)
        {

            break;
        }

        if (col % 2 == 1)
        {
            // row:1~n
            if (row < n)
                row++;
            else
                col++;
        }
        else
        {
            // row: n->1
            if (row > 1)
                row--;
            else
                col++;
        }
    }
    return 0;
}
```

```cpp
    int now = 0;

        if (a[i] > a[1])
            now++;
    now++;
    // 计算第 now 名在第几列、第几行
    int col = (now + (n - 1)) / n; // now 除以 n 上取整
    // 这一列的第几个
    int pos = now - (col - 1) * n;
    // 确定是第几行
    int row;
    if (col % 2 == 1)
        row = pos;
    else
        row = n - pos + 1;
    cout << col << " " << row;
    return 0;
}
```

## C. [CSP-J 2025] 异或和 / xor（民间数据）

### 经济实惠快速的 $60$ 分

```cpp
#include <bits/stdc++.h>
using namespace std;
const int MAXN = 500000;
int n, k;
int a[MAXN + 5];
int sum[MAXN + 5];
// 前 i 个数能凑出来的组数
int f[MAXN + 5];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> k;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    // 处理前缀异或和
    sum[0] = 0;
    for (int i = 1; i <= n; i++)
        sum[i] = sum[i - 1] ^ a[i];
    // dp
    f[0] = 0;
    for (int i = 1; i <= n; i++)
    {
        f[i] = f[i - 1];
        // 找到一个最近的 j 使得 a[j]~a[i] 异或和为 k
        for (int j = i; j >= 1; j--)
            if ((sum[i] ^ sum[j - 1]) == k)
                // a[1]~a[j-1] 的最多数量 + 一组（a[j]~a[i]）
                f[i] = max(f[i], f[j - 1] + 1);
    }
    cout << f[n];
    return 0;
}
```

```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int MAXN = 500000 + 5;
int a[MAXN];
int XOR[MAXN];
int dp[MAXN];
vector<pair<int, int>> vec;
bool cmp(pair<int, int> a, pair<int, int> b) {
    if (a.second != b.second) {
        return a.second > b.second;
    } else { // a.second == b.second
        return a.first > b.first;
    }
}

signed main() {

    int n, k;
    cin >> n >> k;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        XOR[i] = XOR[i - 1] ^ a[i];
    }

    for (int i = 1; i <= n; i++) {
        // 不使用a[i]
        dp[i] = dp[i - 1];
        // 使用a[i]
        int sum = 0;
        int t = i;
        while (t > 0) {
            sum ^= a[t];
            if (sum == k) {
                break;
            }
            t--;
        }
        if (t == 0) { // 没找到包含a[i]的合法区间
        } else {
            dp[i] = max(dp[i], dp[t - 1] + 1);
        }
    }
    cout << dp[n];
    return 0;
}
```

### 不错的正解

```cpp
#include <bits/stdc++.h>
using namespace std;
const int MAXN = 500000;
int n, k;
int a[MAXN + 5];
int sum[MAXN + 5];
// 前 i 个数能凑出来的组数
int f[MAXN + 5];
// pos[i] 记录前缀异或和为 i 的最新下标
int pos[(1 << 20) + 5];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    for (int i = 0; i <= (1 << 20) - 1; i++)
        pos[i] = -1;

    cin >> n >> k;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    // 处理前缀异或和
    sum[0] = 0;
    for (int i = 1; i <= n; i++)
        sum[i] = sum[i - 1] ^ a[i];
    // dp
    pos[0] = 0;
    f[0] = 0;
    for (int i = 1; i <= n; i++)
    {
        f[i] = f[i - 1];
        // 找到一个最近的 j 使得 a[j]~a[i] 异或和为 k
        // (sum[i] ^ sum[j - 1]) == k
        // 等价于 sum[j-1] == sum[i]^k
        if (pos[sum[i] ^ k] != -1)
        {
            int j = pos[sum[i] ^ k] + 1;
            // a[1]~a[j-1] 的最多数量 + 一组（a[j]~a[i]）
            f[i] = max(f[i], f[j - 1] + 1);
        }
        pos[sum[i]] = i;
    }
    cout << f[n];
    return 0;
}
```

```cpp
using namespace std;
const int MAXN = 500000;
int n, k;
int a[MAXN + 5];

        cin >> a[i];
    // 处理前缀异或和
    sum[0] = 0;
    for (int i = 1; i <= n; i++)
        sum[i] = sum[i - 1] ^ a[i];
    // dp
    pos[0] = 0;
    int lastPos = 0; // 上一段的结尾
    int ans = 0;
    for (int i = 1; i <= n; i++)
    {
        if (pos[sum[i] ^ k] + 1 > lastPos)
        {
            ans++;
            lastPos = i;
        }
        pos[sum[i]] = i;
    }
    cout << ans;
    return 0;
}
```

## [CSP-J 2025] 多边形 / polygon（民间数据）

### 经济实惠快速的 $64$ 分

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int MOD = 998244353;
int n, ans;
int a[5005];
// 现在考虑第 now 个数，已经选择了 cnt 个数
// 这些数之和是 sum，最大值是 maxA
void dfs(int now, int cnt, int sum, int maxA)
{
    if (now > n)
    {
        if (sum > 2 * maxA)
            ans++;
        return;
    }
    dfs(now + 1, cnt, sum, maxA);
    dfs(now + 1, cnt + 1, sum + a[now], max(maxA, a[now]));
}
signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    sort(a + 1, a + n + 1);
    if (n == 3)
    {
        if (a[1] + a[2] > a[3])
            cout << 1;
        else
            cout << 0;
    }
    else if (a[n] == 1)
    {
        // 2^n - C(n,0) - C(n,1) - C(n,2)
        int ans = 1;
        for (int i = 1; i <= n; i++)
            ans = ans * 2 % MOD;
        ans -= 1;
        ans -= n;
        ans -= n * (n - 1) / 2;
        ans %= MOD; // -MOD+1 ~ MOD-1
        ans = (ans + MOD) % MOD;
        cout << ans;
    }
    else
    {
        dfs(1, 0, 0, 0);
        cout << ans;
    }
    return 0;
}
```

### 不错的正解

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int MOD = 998244353;
int n, ans;
int a[5005];
// f[i][j] 前 i 个数，和大于 j 的方案数
// 因为所有数都大于 0，所以 f[i][<0] 等价于 f[i][0]
int f[5005][5005];
signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    sort(a + 1, a + n + 1);
    // f[0][~] = 0
    f[0][0] = 1;
    for (int i = 1; i <= n; i++)
        for (int j = 0; j <= 5001; j++)
        {
            f[i][j] = f[i - 1][j] +
                      f[i - 1][max(0LL, j - a[i])];
            f[i][j] %= MOD;
        }
    int ans = 0;
    for (int i = 1; i <= n; i++)
    {
        // a[1]~a[i-1] > a[i]
        // a[1]~a[i-1] >= a[i]+1
        ans += f[i - 1][a[i] + 1];
        ans %= MOD;
    }
    cout << ans;
    return 0;
}
```
